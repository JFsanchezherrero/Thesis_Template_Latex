\newglossaryentry{Anotar}{
    name={Anotar},
    description={Proceso de identificar la localización de los genes y de las regiones codificantes en un genoma y de determinar su función}
}
\newglossaryentry{Barcode}{
    name={Barcode},
    description={Serie de bases conocidas añadidas a una molecula de ADN de interés, por ligación o amplificación, que permite la identificación tras la secuenciación de esta particula.}
}
\newglossaryentry{Biomimesis}{
    name={Biomimesis},
    description={Ciencia que estudia a la naturaleza como fuente de inspiración de nuevas tecnologías innovadoras para resolver aquellos problemas humanos que la naturaleza ha resuelto mediante modelos de sistemas (mecánica), procesos (química) o elementos que imitan o se inspiran en ella}
}
\newglossaryentry{Contig}{
    name={Contig},
    description={Fragmento de ADN obtenido por el ensamblaje de un grupo de reads }
}
\newglossaryentry{Ensamblaje}{
    name={Ensamblaje},
    description={Proceso de descifrar la secuencia genómica a partir de fragmentos secuenciados de ADN}
}
\newglossaryentry{i5K}{
    name={i5K},
    description={El proyecto i5k (5000 arthropod genomes initiative) es una iniciativa global fundada en 2012 para secuenciar las especies mas representativas de artrópodos o con mayor relevancia ecológica, agronómica o biosanitaria}
}
\newglossaryentry{Islas Macaronésicas}{
    name={Islas Macaronésicas},
    description={nombre colectivo de cinco archipiélagos del Atlántico Norte, más o menos cercanos al continente africano: Azores, Canarias, Cabo Verde, Madeira e Islas Salvajes.}
}
\newglossaryentry{K-mer}{
    name={K-mer},
    description={Subcadena de bases dentro de una secuencia de una longitud (k) conocida.}
}
\newglossaryentry{Mapear}{
    name={Mapear},
    description={Proceso de alinear reads a una secuencia de referencia}
}
\newglossaryentry{N50}{
    name={N50},
    description={Longitud de la secuencia en un ensamblaje genomico, donde todas las secuencias están ordenadas por tamaño, donde todos las secuencias mas largas corresponden al 50\% de las bases ensambladas}
}
\newglossaryentry{Primer}{
    name={Primer},
    description={Cebador de ADN empleado durante la amplificación de un fragmento de interés}
}
\newglossaryentry{Read}{
    name={Read},
    description={Secuencia de bases de ADN resultante del proceso de secuenciación}
}
